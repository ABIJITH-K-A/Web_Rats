rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAuth() { return request.auth != null; }
    function uid() { return request.auth.uid; }
    function getRole() { return request.auth.token.role; }

    function isOwner() { return isAuth() && getRole() == "owner"; }
    function isAdmin() { return isAuth() && getRole() in ["owner", "admin"]; }
    function isStaff() { return isAuth() && getRole() in ["owner", "admin", "worker"]; }
    function isWorker() { return isAuth() && getRole() == "worker"; }

    match /users/{userId} {
      allow read: if isAuth();
      allow create: if isAuth() && uid() == userId;
      allow update: if isAuth() && uid() == userId || isAdmin();
      allow delete: if isOwner();
    }

    match /orders/{orderId} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update: if isAuth() && (resource.data.customerId == uid() || isStaff());
      allow delete: if isAdmin();
    }

    match /messages/{messageId} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update, delete: if isAdmin();
    }

    match /uploads/{uploadId} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update, delete: if isAdmin();
    }

    match /revisions/{revisionId} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update: if isStaff();
      allow delete: if isAdmin();
    }

    match /notifications/{notifId} {
      allow read, update: if isAuth() && resource.data.recipientId == uid();
      allow create: if isStaff();
      allow delete: if isAuth() && resource.data.recipientId == uid();
    }

    match /templates/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /templatePurchases/{id} {
      allow read: if isAuth();
      allow write: if false;
    }

    match /inviteKeys/{key} {
      allow read: if isAuth();
      allow create, delete: if isAdmin();
      allow update: if isAuth();
    }

    match /referralCodes/{code} {
      allow read: if isAuth();
      allow write: if false;
    }
  }
}
